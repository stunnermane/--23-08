package me.r8ch.library;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.TextView;
import java.util.List;

import me.r8ch.library.BookListActivity;

public class BookAdapter extends RecyclerView.Adapter<BookAdapter.BookViewHolder> {

    private List<Book> bookList;
    public BookAdapter(List<BookListActivity.Book> bookList) {

    }

    @NonNull
    @Override
    public BookViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View itemView = LayoutInflater.from(parent.getContext())
                .inflate(R.layout.activity_main, parent, false);
        return new BookViewHolder(itemView);
    }

    @Override
    public void onBindViewHolder(@NonNull BookViewHolder holder, int position) {
        Book book = bookList.get(position);
        holder.yearTextView.setText(String.valueOf(book.year));
        holder.titleTextView.setText(book.title);
        holder.authorTextView.setText(book.author);
    }

    @Override
    public int getItemCount() {
        return bookList.size();
    }

    public void updateBookList() {
        updateBookList(null);
    }

    public void updateBookList(List<Book> newBookList) {
        bookList.clear();
        bookList.addAll(newBookList);
        notifyDataSetChanged();
    }

    public static class BookViewHolder extends RecyclerView.ViewHolder {
        public TextView yearTextView;
        public TextView titleTextView;
        public TextView authorTextView;

        public BookViewHolder(View itemView) {
            super(itemView);
            yearTextView = itemView.findViewById(R.id.textView);
            titleTextView = itemView.findViewById(R.id.textView);
            authorTextView = itemView.findViewById(R.id.textView);
        }
    }

    private class Book {
        public char[] year;
        public int title;
        public int author;
    }
}
